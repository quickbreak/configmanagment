```py
s = input()
n = len(s)
print('+', end = '')
for _ in range(n + 2):
    print('-', end = '')
print('+', end = '')

print('\n| ' + s + ' |')

print('+', end = '')
for _ in range(n + 2):
    print('-', end = '')
print('+', end = '')
print()
```
